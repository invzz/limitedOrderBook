cmake_minimum_required(VERSION 3.10)
project(OrderBook)

# Set C++ standard
set(CMAKE_CXX_STANDARD 17)

# Include directories
include_directories(inc)

# Include directories
include_directories(inc/bot)

# 
file(GLOB ORDER_BOOK_SRC "src/*.cc")

file(GLOB BOTS_SRC "src/bot/*.cc")

# Create the OrderBook library
add_library(OrderBookLib ${ORDER_BOOK_SRC} )
add_library(BotsLib ${BOTS_SRC} )

# Find and link spdlog
find_package(spdlog CONFIG REQUIRED)
target_link_libraries(OrderBookLib PRIVATE spdlog::spdlog)
target_link_libraries(BotsLib PRIVATE spdlog::spdlog)

# Create the main application
add_executable(OrderBookApp app/market.cc)
add_compile_definitions(-DOUTPUT_DIR="${CMAKE_CURRENT_SOURCE_DIR}/output/" -DNUM_TICKS=1000 -DTICK_DURATION_MS=0)

# Link the OrderBook library to the application
target_link_libraries(OrderBookApp PRIVATE BotsLib OrderBookLib spdlog::spdlog)

find_package(GTest CONFIG REQUIRED)

# Add tests
enable_testing()
file(GLOB TEST_SOURCES "test/*.cc")


foreach(TEST_SOURCE ${TEST_SOURCES})
    # Get the filename without the path or extension to name the executable
    get_filename_component(TEST_NAME ${TEST_SOURCE} NAME_WE)

    # Add an executable for each test source file
    add_executable(${TEST_NAME} ${TEST_SOURCE})

    # Link each test executable to the OrderBook library and GTest
    target_link_libraries(${TEST_NAME} PRIVATE BotsLib OrderBookLib GTest::gtest)

    # Link spdlog to each test executable if needed
    target_link_libraries(${TEST_NAME} PRIVATE spdlog::spdlog)

    # Register each test with CTest
    add_test(NAME ${TEST_NAME} COMMAND ${TEST_NAME})
endforeach()
